52inline auto nonosc_step(SolverInfo &&info, Scalar x0, Scalar h, YScalar y0,
53 YScalar dy0, Scalar epsres, Allocator &&alloc) {
54 using complex_t = std::complex<Scalar>;
55
56 Scalar maxerr = 10 * epsres;
57 auto N = info.nini_;
58 auto Nmax = info.nmax_;
59 auto cheby = spectral_chebyshev(info, x0, h, y0, dy0, 0, alloc);
60 auto yprev = std::get<0>(cheby);
61 auto dyprev = std::get<1>(cheby);
62 auto xprev = std::get<2>(cheby);
63 int iter = 0;
64 while (maxerr > epsres) {
65 iter++;
66 N *= 2;
67 if (N > Nmax) {
68 return std::make_tuple(false, complex_t(0.0, 0.0), complex_t(0.0, 0.0),
69 maxerr, yprev, dyprev);
70 }
71 auto cheb_num = static_cast<int>(std::log2(N / info.nini_));
72 auto cheby2 = spectral_chebyshev(info, x0, h, y0, dy0, cheb_num, alloc);
73 auto y = std::get<0>(std::move(cheby2));
74 auto dy = std::get<1>(std::move(cheby2));
75 auto x = std::get<2>(std::move(cheby2));
76 maxerr = std::abs((yprev(0) - y(0)) / y(0));
77 if (std::isnan(maxerr)) {
78 maxerr = std::numeric_limits<Scalar>::infinity();
79 }
80 yprev = std::move(y);
81 dyprev = std::move(dy);
82 xprev = std::move(x);
83 }
84 return std::make_tuple(true, yprev(0), dyprev(0), maxerr, yprev, dyprev);
85}

139template <typename SolverInfo, typename OmegaVec, typename GammaVec,

141inline auto osc_step(SolverInfo &&info, OmegaVec &&omega_s, GammaVec &&gamma_s,
142 Scalar x0, Scalar h, YScalar y0, YScalar dy0,
143 Scalar epsres, Allocator &&alloc) {
144 using complex_t = std::complex<Scalar>;
145 using vectorc_t = vector_t<complex_t>;
146 bool success = true;
147 auto &&Dn = info.Dn();
148 auto y = eval(alloc, complex_t(0.0, 1.0) * omega_s);
149 auto delta = [&](const auto &r, const auto &y) {
150 return (-r.array() / (2.0 * (y.array() + gamma_s.array()))).matrix().eval();
151 };
152 auto R = [&](const auto &d) {
153 return 2.0 / h * (Dn * d) + d.array().square().matrix();
154 };
155 auto Ry = (complex_t(0.0, 1.0) * 2.0
156 * (1.0 / h * (Dn * omega_s) + gamma_s.cwiseProduct(omega_s)))
157 .eval();
158 Scalar maxerr = Ry.array().abs().maxCoeff();
159
160 arena_matrix<vectorc_t> deltay(alloc, Ry.size(), 1);
161 Scalar prev_err = std::numeric_limits<Scalar>::infinity();
162 while (maxerr > epsres) {
163 deltay = delta(Ry, y);
164 y += deltay;
165 Ry = R(deltay);
166 maxerr = Ry.array().abs().maxCoeff();
167 if (maxerr >= prev_err) {
168 success = false;
169 break;
170 }
171 prev_err = maxerr;
172 }
173 if (info.denseout_) {
174 auto u1 = eval(alloc, h / 2.0 * (info.integration_matrix_ * y));
175 auto f1 = eval(alloc, (u1).array().exp().matrix());
176 auto f2 = eval(alloc, f1.conjugate());
177 auto du2 = eval(alloc, y.conjugate());
178 auto ap_top = (dy0 - y0 * du2(du2.size() - 1));
179 auto ap_bottom = (y(y.size() - 1) - du2(du2.size() - 1));
180 auto ap = ap_top / ap_bottom;
181 auto am = (dy0 - y0 * y(y.size() - 1))
182 / (du2(du2.size() - 1) - y(y.size() - 1));
183 auto y1 = eval(alloc, ap * f1 + am * f2);
184 auto dy1
185 = eval(alloc, ap * y.cwiseProduct(f1) + am * du2.cwiseProduct(f2));
186 Scalar phase = std::imag(f1(0));
187 return std::make_tuple(success, y1(0), dy1(0), maxerr, phase, u1,
188 std::make_pair(ap, am));
189 } else {
190 complex_t f1 = std::exp(h / 2.0 * (info.quadwts_.dot(y)));
191 auto f2 = std::conj(f1);
192 auto du2 = y.conjugate().eval();
193 auto ap_top = (dy0 - y0 * du2(du2.size() - 1));
194 auto ap_bottom = (y(y.size() - 1) - du2(du2.size() - 1));
195 auto ap = ap_top / ap_bottom;
196 auto am = (dy0 - y0 * y(y.size() - 1))
197 / (du2(du2.size() - 1) - y(y.size() - 1));
198 auto y1 = (ap * f1 + am * f2);
199 auto dy1 = (ap * y * f1 + am * du2 * f2).eval();
200 Scalar phase = std::imag(f1);
201 return std::make_tuple(success, y1, dy1(0), maxerr, phase,
202 arena_matrix<vectorc_t>(alloc, y.size(), 0),
203 std::make_pair(ap, am));
204 }
205}
