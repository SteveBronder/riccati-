18inline Eigen::VectorXd logspace(double start, double end, int num,
19 double base) {
20 Eigen::VectorXd result(num);
21 double delta = (end - start) / (num - 1);
22
23 for (int i = 0; i < num; ++i) {
24 result[i] = std::pow(base, start + i * delta);
25 }
26
27 return result;
28}
